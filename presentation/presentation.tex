\begin{itemize}
\item Sanskrit to Sanskrit dictionary
\item Sanskrit to English dictionary
\item English to Sanskrit dictionary
\end{itemize}
Dictionary words creation
\begin{itemize}

\begin{itemize}

\end{itemize}
\end{frame}
